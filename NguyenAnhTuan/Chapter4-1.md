```java
public class Calc {
   
   static public int add (int a, int b) {
      return a + b;
   }
   
   static public int subtract (int a, int b) {
      return a - b;
   }
   
   static public int multiply (int a, int b) {
      return a * b;
   }
   
   static public int divide (int a, int b) {
      return a / b;
   }
```
